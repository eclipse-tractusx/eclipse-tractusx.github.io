export const MasterDataOverview = () => {
  // Collect all KITs from different sections
  const foundationKits = [...(kitsData.dataspaceFoundation || [])];
  const industryCoreKits = [...(kitsData.industryCoreFoundation || [])];
  const useCaseKits = [...(kitsData.useCases || [])];
  
  // Collect all industry-specific KITs
  const industrySpecificKits = [];
  if (kitsData.industryKits) {
    Object.values(kitsData.industryKits).forEach(kitsArray => {
      if (Array.isArray(kitsArray)) {
        industrySpecificKits.push(...kitsArray);
      }
    });
  }

  const allKits = [...foundationKits, ...industryCoreKits, ...useCaseKits, ...industrySpecificKits];

  const totalKits = allKits.length;
  const graduatedKits = allKits.filter(kit => kit.maturity?.currentLevel === 'Graduated').length;
  const incubationKits = allKits.filter(kit => kit.maturity?.currentLevel === 'Incubating').length;
  const sandboxKits = allKits.filter(kit => kit.maturity?.currentLevel === 'Sandbox').length;
  const deprecatedKits = allKits.filter(kit => kit.deprecated).length;
  const newKits = allKits.filter(kit => kit.metadata?.new).length;

  // Get unique industries
  const allIndustries = [...new Set(allKits.flatMap(kit => kit.industries || []))];
  
  // Get industry names that have specific KITs
  const industryNames = kitsData.industryKits ? Object.keys(kitsData.industryKits) : [];
  
  return (
    <div>
      {/* Statistics Overview */}
      <div style={{ 
        display: 'flex', 
        gap: '16px', 
        marginBottom: '40px',
        flexWrap: 'wrap',
        justifyContent: 'center'
      }}>
        <KitStatsCard title="Total KITs" count={totalKits} color="#3b82f6" IconComponent={InventoryIcon} />
        <KitStatsCard title="Graduated" count={graduatedKits} color="#10b981" IconComponent={SchoolIcon} />
        <KitStatsCard title="Incubation" count={incubationKits} color="#f59e0b" IconComponent={CachedIcon} />
        <KitStatsCard title="Sandbox" count={sandboxKits} color="#6b7280" IconComponent={ScienceIcon} />
        <KitStatsCard title="New KITs" count={newKits} color="#8b5cf6" IconComponent={AutoAwesomeIcon} />
        <KitStatsCard title="Industries" count={allIndustries.length} color="#ec4899" IconComponent={PublicIcon} />
      </div>

      {/* Industry Master Data */}
      {industries && industries.length > 0 && (
        <IndustrySection 
          title="Industries"
          industries={industries}
          color="#ec4899"
        />
      )}

      {/* Categories */}
      {kitsData.dataspaceFoundation && (
        <CategorySection 
          title="Dataspace Foundation KITs"
          kits={kitsData.dataspaceFoundation}
          color="#3b82f6"
        />
      )}
      
      {kitsData.industryCoreFoundation && (
        <CategorySection 
          title="Industry Core Foundation KITs"
          kits={kitsData.industryCoreFoundation}
          color="#3b82f6"
        />
      )}
      
      {kitsData.useCases && (
        <CategorySection 
          title="Cross Dataspace Use Cases KITs"
          kits={kitsData.useCases}
          color="#10b981"
        />
      )}

      {/* Industry Specific KITs */}
      {kitsData.industryKits && Object.entries(kitsData.industryKits).some(([_, kits]) => kits && kits.length > 0) && (
        <div style={{ marginTop: '40px' }}>
          <h2 style={{ 
            paddingBottom: '12px',
            marginBottom: '30px',
            fontSize: '28px'
          }}>
            Industry Specifics
          </h2>
          {Object.entries(kitsData.industryKits).map(([industryName, kits]) => (
            kits && kits.length > 0 && (
              <CategorySection 
                key={industryName}
                title={`${industryName.charAt(0).toUpperCase() + industryName.slice(1)} Specific KITs`}
                kits={kits}
                color="#f59e0b"
              />
            )
          ))}
        </div>
      )}
    </div>
  );
};

## Industry & Dataspace Master Data Overview

This page provides a comprehensive overview of all industries and dataspaces in the Tractus-X ecosystem. The master data is maintained in the [`kitsData.js`](../data/kitsData.js) file and serves as the foundation for organizing KITs and dataspace implementations.

### Ecosystem Hierarchy

The master data follows this organizational hierarchy:

```
Dataspace Foundation
└── Industry Core KITs
  └── Cross-Dataspace Use Cases
    └── Industry-Specific KITs
      ├── Automotive Industry
      │   └── Dataspaces
      │       └── Catena-X
      ├── Shop Floor Industry
      │   └── Dataspaces
      │       └── <Future Implementations>
      └── ... (other industries)
``` 

### What is Master Data?

The master data includes two primary components that structure the entire ecosystem:

#### Industry Master Data
Core information about each industry vertical:
- **Identity**: Unique ID, name, and subtitle
- **Visual Branding**: Icon component, colors, and gradient styling
- **Content**: Description and purpose within the ecosystem
- **Associated Dataspaces**: Array of dataspace implementations within this industry

#### Dataspace Master Data  
Essential information about each dataspace implementation:
- **Identity**: Unique ID, name, and subtitle
- **Branding**: Logo (SVG/PNG), colors, and gradient styling (**MUST** be hosted by the dataspace owner)
- **Content**: Detailed description and purpose
- **Links**: Official website URL for the dataspace
- **Industry Association**: Which industry vertical this dataspace belongs to

## Managing Master Data

### Adding a New Industry

To add a new industry to the master data:

**Required fields:**
- `id`, `name`, `subtitle`, `description`
- `icon` (Material-UI component)
- `colors` (layer1-4), `gradient`
- `url`, `dataspaces` (array)

Each industry entry follows this structure:

```javascript
{
  id: 'unique-industry-id',
  name: 'Industry Name',
  subtitle: 'Brief industry tagline',
  description: 'Generic description focusing on industry context and purpose',
  icon: MaterialUIIconComponent,
  gradient: 'linear-gradient(135deg, #color1, #color2, #color3)',
  url: '/docs-kits/industry/industry-name',
  dataspaces: [
    /* array of dataspace objects (described below)*/
  ]
}
```

### Adding a New Dataspace

To add a new dataspace to the master data:

**Required fields:**
- `name`, `url`, `subtitle`, `description`
- `logo` (object with src, alt, width, height)
- `colors` (layer1-4), `gradient`



Each dataspace entry follows this structure:

```javascript
{
  id: 'dataspace-id',
  name: 'Dataspace Name',
  subtitle: 'Industry Focus',
  description: 'Detailed description of the dataspace purpose and goals',
  logo: {
    src: 'https://example.com/logo.svg',
    alt: 'Logo Alt Text',
    width: 80,
    height: 'auto'
  },
  colors: {
    layer1: '#color1',
    layer2: '#color2',
    layer3: '#color3',
    layer4: '#color4'
  },
  gradient: 'linear-gradient(135deg, #color1, #color2)',
  url: 'https://dataspace-website.com'
}
```
