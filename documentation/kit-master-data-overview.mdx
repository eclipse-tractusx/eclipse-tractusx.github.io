export const MasterDataOverview = () => {
  // Collect all KITs from different sections
  const foundationKits = [...(kitsData.dataspaceFoundation || [])];
  const industryCoreKits = [...(kitsData.industryCoreFoundation || [])];
  const useCaseKits = [...(kitsData.useCases || [])];
  
  // Collect all dataspace-specific KITs
  const dataspaceSpecificKits = [];
  if (kitsData.dataspaceKits) {
    Object.values(kitsData.dataspaceKits).forEach(kitsArray => {
      if (Array.isArray(kitsArray)) {
        dataspaceSpecificKits.push(...kitsArray);
      }
    });
  }

  const allKits = [...foundationKits, ...industryCoreKits, ...useCaseKits, ...dataspaceSpecificKits];

  const totalKits = allKits.length;
  const graduatedKits = allKits.filter(kit => kit.maturity?.currentLevel === 'Graduated').length;
  const incubationKits = allKits.filter(kit => kit.maturity?.currentLevel === 'Incubating').length;
  const sandboxKits = allKits.filter(kit => kit.maturity?.currentLevel === 'Sandbox').length;
  const deprecatedKits = allKits.filter(kit => kit.deprecated).length;
  const newKits = allKits.filter(kit => kit.metadata?.new).length;

  // Get unique dataspaces
  const allDataspaces = [...new Set(allKits.flatMap(kit => kit.dataspaces || []))];
  
  // Get dataspace names that have specific KITs
  const dataspaceNames = kitsData.dataspaceKits ? Object.keys(kitsData.dataspaceKits) : [];
  
  return (
    <div>
      {/* Statistics Overview */}
      <div style={{ 
        display: 'flex', 
        gap: '16px', 
        marginBottom: '40px',
        flexWrap: 'wrap',
        justifyContent: 'center'
      }}>
        <KitStatsCard title="Total KITs" count={totalKits} color="#3b82f6" IconComponent={InventoryIcon} />
        <KitStatsCard title="Graduated" count={graduatedKits} color="#10b981" IconComponent={SchoolIcon} />
        <KitStatsCard title="Incubation" count={incubationKits} color="#f59e0b" IconComponent={CachedIcon} />
        <KitStatsCard title="Sandbox" count={sandboxKits} color="#6b7280" IconComponent={ScienceIcon} />
        <KitStatsCard title="New KITs" count={newKits} color="#8b5cf6" IconComponent={AutoAwesomeIcon} />
        <KitStatsCard title="Dataspaces" count={allDataspaces.length} color="#ec4899" IconComponent={PublicIcon} />
      </div>

      {/* Dataspace Master Data */}
      {dataspaces && dataspaces.length > 0 && (
        <DataspaceSection 
          title="Dataspaces"
          dataspaces={dataspaces}
          color="#ec4899"
        />
      )}

      {/* Categories */}
      {kitsData.dataspaceFoundation && (
        <CategorySection 
          title="Dataspace Foundation KITs"
          kits={kitsData.dataspaceFoundation}
          color="#3b82f6"
        />
      )}
      
      {kitsData.industryCoreFoundation && (
        <CategorySection 
          title="Industry Core Foundation KITs"
          kits={kitsData.industryCoreFoundation}
          color="#3b82f6"
        />
      )}
      
      {kitsData.useCases && (
        <CategorySection 
          title="Cross Dataspace Use Cases KITs"
          kits={kitsData.useCases}
          color="#10b981"
        />
      )}

      {/* Dataspace Specific KITs */}
      {kitsData.dataspaceKits && Object.entries(kitsData.dataspaceKits).some(([_, kits]) => kits && kits.length > 0) && (
        <div style={{ marginTop: '40px' }}>
          <h2 style={{ 
            paddingBottom: '12px',
            marginBottom: '30px',
            fontSize: '28px'
          }}>
            Dataspace Specifics
          </h2>
          {Object.entries(kitsData.dataspaceKits).map(([dataspaceName, kits]) => (
            kits && kits.length > 0 && (

                title={`${dataspaceName.toUpperCase()} Specific KITs`}

                color="#f59e0b"
              />
            )
          ))}
        </div>
      )}
    </div>
  );

## KIT Master Data Overview

This page provides a comprehensive overview of all KITs (Keep It Together) registered in the Tractus-X ecosystem. The master data is maintained in the [`kitsData.js`](../data/kitsData.js) file and includes information about maturity levels, dataspaces, versions, and more.

### What is Master Data?

The master data includes two main components:

#### KIT Master Data
Essential metadata for each KIT including:
- **Identification**: Unique ID, name, and route
- **Visual Elements**: Logo, colors, and styling
- **Maturity Information**: Current level (Sandbox, Incubation, Graduated) and graduation status
- **Dataspace Association**: Which dataspaces use this KIT
- **Versioning**: Creation date, last update, and current version
- **Status Flags**: New, deprecated, or active status

#### Dataspace Master Data  
Information about each dataspace in the Tractus-X ecosystem:
- **Identity**: Unique ID, name, and subtitle
- **Branding**: Logo, colors, and gradient styling
- **Content**: Description and purpose
- **Links**: Official website URL

## Managing Master Data

### Adding a New KIT

To add a new KIT to the master data:

1. **Choose the correct category**: `dataspaceFoundation`, `industryCoreFoundation`, `useCase`, or `dataspaceSpecific`
2. **Import the logo**: Add SVG logo to `/static/img/kits/<kit-name>/` and import it
3. **Follow the data structure**: Use the template above with all required fields
4. **Set appropriate maturity level**: Start with "Sandbox" for new KITs, and set the new flag if applicable
5. **Add dataspace associations**: Specify which dataspaces use this KIT

### Updating KIT Information

When updating KIT information:

- **Update `lastUpdated`**: Change to current date when making modifications
- **Version bumping**: Update `latestVersion` when KIT content changes significantly
- **Maturity progression**: Update `currentLevel` when KIT graduates to next level
- **Deprecation**: Set `deprecated: true` for KITs no longer maintained

### Adding a New Dataspace

To add a new dataspace to the master data:

1. **Add to dataspaces array**: Include complete dataspace information
2. **Set up branding**: Logo, colors, and gradient for visual identity
3. **Create dataspace KIT section**: Add empty array in `dataspaceKits` object
4. **Update documentation**: Include dataspace in relevant documentation

### Managing Dataspace KITs

When managing dataspace-specific KITs:

- **Add to correct dataspace**: Use the appropriate key in `dataspaceKits`
- **Cross-dataspace compatibility**: KITs can be used by multiple dataspaces
- **Dataspace association**: Update the `dataspaces` array in KIT metadata

### Data Structure

Each KIT entry follows this structure:

```javascript
{
  id: 'unique-kit-id',
  name: 'KIT NAME IN UPPERCASE',
  logo: KitLogoComponent,
  logoHeight: 80,
  logoWidth: 80,
  route: '/docs-kits/kits/kit-name/adoption-view',
  colors: {
    primary: '#hexcolor',
    gradient: 'linear-gradient(...)'
  },
  maturity: {
    currentLevel: 'Graduated|Incubation|Sandbox',
    graduationStatus: 'Draft|In Progress|In Review|No Further Development'
  },
  deprecated: false,
  dataspaces: ['Catena-X', 'Factory-X'],
  description: 'Kit description text',
  metadata: {
    created: 'YYYY-MM-DD',
    lastUpdated: 'YYYY-MM-DD',
    latestVersion: 'x.y.z',
    new: false
  }
}
```
